        // N/A checkbox functions

        function setFieldValue(fieldId, value, checkbox) {

            const field = document.getElementById(fieldId);

            if (checkbox.checked) {

                field.value = value;

                field.disabled = true;

            } else {

                field.value = '';

                field.disabled = false;

            }

        }

        // Handle form submission - IMMEDIATE SUCCESS

        document.getElementById('patientForm').addEventListener('submit', function(e) {

            e.preventDefault();

            

            // Get patient ID

            const patientId = document.getElementById('patientId').value;

            

            // Collect form data for background submission

            const evolutionSymptoms = [];

            document.querySelectorAll('input[name="evolution"]:checked').forEach(checkbox => {

                evolutionSymptoms.push(checkbox.value);

            });

            

            const pregnancyStatus = [];

            document.querySelectorAll('input[name="pregnancy"]:checked').forEach(checkbox => {

                pregnancyStatus.push(checkbox.value);

            });

            

            const rosSymptoms = [];

            document.querySelectorAll('input[name="ros"]:checked').forEach(checkbox => {

                rosSymptoms.push(checkbox.value);

            });

            

            if (document.getElementById('otherSymptoms').value) {

                rosSymptoms.push(document.getElementById('otherSymptoms').value);

            }

            

            const dateValue = document.getElementById('date-na').checked ? 'N/A' : document.getElementById('startDate').value;

            

            const formData = {

                patientId: patientId,

                gender: document.getElementById('gender').value,

                age: document.getElementById('age').value,

                chiefComplaint: document.getElementById('chiefComplaint').value,

                startDate: dateValue,

                trigger: document.getElementById('trigger').value || '',

                location: document.getElementById('location').value,

                description: document.getElementById('description').value,

                severity: document.getElementById('severitySlider').value,

                evolution: document.getElementById('evolution-na-check').checked ? 'N/A' : evolutionSymptoms.join(', '),

                worsening: document.getElementById('worsening').value || '',

                relief: document.getElementById('relief').value || '',

                associatedSymptoms: rosSymptoms.join(', ') || '',

                chronicConditions: document.getElementById('chronicConditions').value || '',

                allergies: document.getElementById('allergies').value || '',

                pregnantBreastfeeding: document.getElementById('pregnancy-na-check').checked ? 'N/A' : pregnancyStatus.join(', '),

                treatments: document.getElementById('treatments').value || '',

                treatmentEffectiveness: document.getElementById('treatmentEffectiveness').value || '',

                additionalInfo: document.getElementById('additionalInfo').value || ''

            };

            

            // IMMEDIATELY show success page

            showSuccessPage(patientId);

            

            // Submit form data in background (no await - fire and forget)

            fetch('/submit-form', {

                method: 'POST',

                headers: {

                    'Content-Type': 'application/json'

                },

                body: JSON.stringify(formData)

            }).catch(error => {

                // Silent error handling - don't interrupt user experience

                console.error('Background submission error:', error);

            });

        });

    </script>
